          GROUP BY p.product_id, o.utm_source_last
      ),
      ranked AS (
          SELECT

              channel,
              orders_cnt,
              row_number() OVER (PARTITION BY product_id ORDER BY orders_cnt DESC) AS rn
          FROM channel_counts
      )
      SELECT product_id, channel, orders_cnt
      FROM ranked
      WHERE rn = 1
